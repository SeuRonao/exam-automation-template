\element{example-lua}{
	\begin{question}{example-lua-random}
		\luadirect{a = math.random(0, 10)}
		\luadirect{b = math.random(0, 10)}
		How much is \(\luadirect{tex.print(a)} + \luadirect{tex.print(b)}\)?
		\begin{choiceshoriz}
			\wrongchoice{\luadirect{tex.print(a+b-2)}}
			\wrongchoice{\luadirect{tex.print(a+b-1)}}
			\correctchoice{\luadirect{tex.print(a+b)}}
			\wrongchoice{\luadirect{tex.print(a+b+1)}}
			\wrongchoice{\luadirect{tex.print(a+b+2)}}
		\end{choiceshoriz}
		\explain{The value is \(\luadirect{tex.print(a+b)}\).}
	\end{question}
}

\element{example-lua}{
	\begin{questionmultx}{example-lua-random-open}

	\end{questionmultx}
}
